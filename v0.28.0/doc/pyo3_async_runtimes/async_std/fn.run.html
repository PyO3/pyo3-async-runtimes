fn main() {
    // call this or use pyo3 0.14 "auto-initialize" feature
    Python::initialize();

    Python::attach(|py| {
        pyo3_async_runtimes::async_std::run(py, async move {
            async_std::task::sleep(Duration::from_secs(1)).await;
            Ok(())
        })
        .map_err(|e| {
            e.print_and_set_sys_last_vars(py);
        })
        .unwrap();
    })
}
