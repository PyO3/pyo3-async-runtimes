54enum Pyo3Runtime {

58impl Deref for Pyo3Runtime {
59    type Target = Runtime;
60    fn deref(&self) -> &Self::Target {
61        match self {
62            Self::Borrowed(rt) => rt,
63            Self::Owned(rt) => rt,
64        }
65    }

100impl ContextExt for TokioRuntime {
101    fn scope<F, R>(locals: TaskLocals, fut: F) -> Pin<Box<dyn Future<Output = R> + Send>>
102    where
103        F: Future<Output = R> + Send + 'static,
104    {
105        let cell = UnsyncOnceCell::new();
106        cell.set(locals).unwrap();
107
108        Box::pin(TASK_LOCALS.scope(cell, fut))
109    }
110
111    fn get_task_locals() -> Option<TaskLocals> {
112        TASK_LOCALS
113            .try_with(|c| {
114                c.get()
115                    .map(|locals| Python::with_gil(|py| locals.clone_ref(py)))
116            })
117            .unwrap_or_default()
118    }
119}

188/// Get a reference to the current tokio runtime
189pub fn get_runtime<'a>() -> &'a Runtime {
190    TOKIO_RUNTIME.get_or_init(|| {

192            .lock()
193            .unwrap()
194            .build()
195            .expect("Unable to build Tokio runtime");
196        Pyo3Runtime::Owned(rt)

275/// Convert a Rust Future into a Python awaitable
276///
277/// If the `asyncio.Future` returned by this conversion is cancelled via `asyncio.Future.cancel`,
278/// the Rust future will be cancelled as well (new behaviour in `v0.15`).
279///
280/// Python `contextvars` are preserved when calling async Python functions within the Rust future
281/// via [`into_future`] (new behaviour in `v0.15`).
282///
283/// > Although `contextvars` are preserved for async Python functions, synchronous functions will
284/// > unfortunately fail to resolve them when called within the Rust future. This is because the
285/// > function is being called from a Rust thread, not inside an actual Python coroutine context.
286/// >
287/// > As a workaround, you can get the `contextvars` from the current task locals using
288/// > [`get_current_locals`] and [`TaskLocals::context`](`crate::TaskLocals::context`), then wrap your
289/// > synchronous function in a call to `contextvars.Context.run`. This will set the context, call the
290/// > synchronous function, and restore the previous context when it returns or raises an exception.
291///
292/// # Arguments
293/// * `py` - PyO3 GIL guard
294/// * `locals` - The task locals for the given future
295/// * `fut` - The Rust future to be converted
296///
297/// # Examples
298///
299/// ```
300/// use std::time::Duration;
301///
302/// use pyo3::prelude::*;
303///
304/// /// Awaitable sleep function
305/// #[pyfunction]
306/// fn sleep_for<'p>(py: Python<'p>, secs: Bound<'p, PyAny>) -> PyResult<Bound<'p, PyAny>> {
307///     let secs = secs.extract()?;
308///     pyo3_async_runtimes::tokio::future_into_py_with_locals(
309///         py,
310///         pyo3_async_runtimes::tokio::get_current_locals(py)?,
311///         async move {
312///             tokio::time::sleep(Duration::from_secs(secs)).await;
313///             Python::with_gil(|py| Ok(py.None()))
314///         }
315///     )
316/// }
317/// ```
318pub fn future_into_py_with_locals<F, T>(
319    py: Python,
320    locals: TaskLocals,
321    fut: F,
322) -> PyResult<Bound<PyAny>>
323where
324    F: Future<Output = PyResult<T>> + Send + 'static,
325    T: for<'py> IntoPyObject<'py>,
326{
327    generic::future_into_py_with_locals::<TokioRuntime, F, T>(py, locals, fut)
328}

548/// Convert a Python `awaitable` into a Rust Future
549///
550/// This function converts the `awaitable` into a Python Task using `run_coroutine_threadsafe`. A
551/// completion handler sends the result of this Task through a
552/// `futures::channel::oneshot::Sender<PyResult<PyObject>>` and the future returned by this function
553/// simply awaits the result through the `futures::channel::oneshot::Receiver<PyResult<PyObject>>`.
554///
555/// # Arguments
556/// * `awaitable` - The Python `awaitable` to be converted
557///
558/// # Examples
559///
560/// ```
561/// use std::time::Duration;
562/// use std::ffi::CString;
563///
564/// use pyo3::prelude::*;
565///
566/// const PYTHON_CODE: &'static str = r#"
567/// import asyncio
568///
569/// async def py_sleep(duration):
570///     await asyncio.sleep(duration)
571/// "#;
572///
573/// async fn py_sleep(seconds: f32) -> PyResult<()> {
574///     let test_mod = Python::with_gil(|py| -> PyResult<PyObject> {
575///         Ok(
576///             PyModule::from_code(
577///                 py,
578///                 &CString::new(PYTHON_CODE).unwrap(),
579///                 &CString::new("test_into_future/test_mod.py").unwrap(),
580///                 &CString::new("test_mod").unwrap(),
581///             )?
582///             .into()
583///         )
584///     })?;
585///
586///     Python::with_gil(|py| {
587///         pyo3_async_runtimes::tokio::into_future(
588///             test_mod
589///                 .call_method1(py, "py_sleep", (seconds.into_pyobject(py).unwrap(),))?
590///                 .into_bound(py),
591///         )
592///     })?
593///     .await?;
594///     Ok(())
595/// }
596/// ```
597pub fn into_future(
598    awaitable: Bound<PyAny>,
599) -> PyResult<impl Future<Output = PyResult<PyObject>> + Send> {
600    generic::into_future::<TokioRuntime>(awaitable)
